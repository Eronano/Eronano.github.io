A - Print a Pedestal (Codeforces logo?)

题意

有一个形似颁奖台的积木（？），要求第 2-th 严格低于 1-st，同时 3-rd 严格低于 2-th 的高度。

三个高度不能有任意两两相等，于是把三个高度分别从 2，3，1开始增加，增加的优先级 1-st > 2-nd > 3-rd。

Code 
#include <iostream>
#include <ctime>
#include <vector>
#define int long long
using namespace std;
void solve() {
    int n;
    cin >> n;
    int high = (n - 6) / 3;
    if((n - 6) % 3 == 1)
        cout << 2 + high << ' ' << 4 + high << ' ' << 1 + high << '\n';
    else if((n - 6) % 3 == 2)
        cout << 3 + high << ' ' << 4 + high << ' ' << 1 + high << '\n';
    else cout << 2 + high << ' ' << 3 + high << ' ' << 1 + high << '\n';
}
//=====================================
signed main() {
#ifdef LOCAL
    freopen("in.in", "r", stdin);
    freopen("out.out", "w", stdout);
    clock_t c1 = clock();
#endif
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    //=====================================
    int t;
    cin >> t;
    while(t--) solve();
    //=====================================
#ifdef LOCAL
    cerr << "Time used: " << clock() - c1 << " ms" << '\n';
#endif
    return 0;
}

B - Array Decrements

题意

有两个数组 aaa 和 bbb，每次操作可以将 aaa 中每个元素替换成 ai−1a_i - 1ai​−1（000除外），问你是否能通过进行一系列操作使得数组 aaa 和 数组 bbb 相等。

记录下所有 aia_iai​ 和 bib_ibi​ 的差，再以其中的最大值对每一个差值进行比较，如果某一个差小于这个最大值且 bib_ibi​ 不为 000，那必然是不成立的。另外题目好像还没有规定 bi ≤ aib_i~\leq~a_ibi​ ≤ ai​，感觉得再判断一下。

Code 
#include <iostream>
#include <ctime>
#include <vector>
#define int long long
using namespace std;
void solve() {
    int n, mx = -1;
    cin >> n;
    vector<int> v(n), b(n), val(n);
    for(auto &it : v) cin >> it;
    for(auto &it : b) cin >> it;
    for(int i = 0; i < n; ++i){
        val[i] = v[i] - b[i];
        mx = max(mx, val[i]);
        if(val[i] < 0){
            cout << "NO" << '\n';
            return;
        }
    }
    for(int i = 0; i < n; ++i){
        if(val[i] < mx && b[i]){
            cout << "NO" << '\n';
            return;
        }
    }
    cout << "YES" << '\n';
}
//=====================================
signed main() {
#ifdef LOCAL

    freopen("out.out", "w", stdout);
    clock_t c1 = clock();
#endif

    //=====================================

    return 0;
}

C - Restoring the Duration of Tasks

题意

有 nnn 个任务，分别告诉你它们的开始时间和完成时间，当然当前任务没有完成时不能开始下一任务，当前任务已完成时需等待下一任务开始，询问每个任务的执行时间。

模拟，第一个任务时间可以直接计算，后面任务时间需要将起始时间与上一个任务结束时间进行比较得出

Code 
#include <iostream>
#include <ctime>
#include <vector>
#define int long long
using namespace std;
void solve() {
    int n;
    cin >> n;
    vector<int> b(n), f(n);
    for(auto &it : b) cin >> it;
    for(auto &it : f) cin >> it;
    cout << f[0] - b[0] << ' ';
    for(int i = 1; i < n; ++i){
        if(f[i - 1] <= b[i]) cout << f[i] - b[i] << ' ';
        else cout << f[i] - f[i - 1] << ' ';
    }
    cout << '\n';
}
//=====================================

D - Black and White Stripe

题意

一个长度为 nnn 的字符串，只包含 BBB 和 WWW，每次操作可以将一个 WWW 变成 BBB ，问最少进行多少次操作可以得到连续的 kkk 个 BBB。

写累了…总之用一个前缀和数组记录直到当前位置把 WWW 变成 BBB 的代价，得到连续的 kkk 个 BBB的最小代价就只需要计算 pre[i]−pre[i−k]pre[i] - pre[i - k]pre[i]−pre[i−k]。

Code 
#include <iostream>
#include <ctime>
#include <vector>
#define int long long
using namespace std;
void solve() {
    int n, k;
    string s;
    cin >> n >> k >> s;
    vector<int> pre(n + 1);
    for(int i = 1; i <= n; ++i){
        pre[i] += pre[i - 1];
        if(s[i - 1] == 'W') pre[i]++;
    }
    int ans = INT_MAX;
    for(int i = k; i <= n; ++i) ans = min(ans, pre[i] - pre[i - k]);
    cout << ans << '\n';
}
//=====================================
signed main() {
#ifdef LOCAL
    freopen("in.in", "r", stdin);
    freopen("out.out", "w", stdout);
    clock_t c1 = clock();
#endif
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    //=====================================
    int t;
    cin >> t;
    while(t--) solve();
    //=====================================
#ifdef LOCAL
    cerr << "Time used: " << clock() - c1 << " ms" << '\n';
#endif
    return 0;
}

E - Price Maximization

题意

给定一个数组 aaa 和 kkk，将数组中的数两两打包使得价值 ⌊ai+ajk⌋\left \lfloor\frac{a_i + a_j}{k} \right \rfloor⌊kai​+aj​​⌋ 最大。

为了使价值最大需要把 aia_iai​ 对于 kkk 的余数利用起来，尽可能多的利用更大的余数，搭配一个较小的余数产生一个大于等于 kkk 的和，这样就可以使价值达到最大，因为一次最多只能使其价值加一。

Code 
#include <iostream>
#include <ctime>
#include <vector>
#include <algorithm>
#define int long long
using namespace std;
void solve() {
    int n, k, ans = 0;
    cin >> n >> k;
    vector<int> v(n + 1);
    for(int i = 1; i <= n; ++i){
        cin >> v[i];
        ans += v[i] / k;
        v[i] = v[i] % k;
    }
    sort(v.begin() + 1, v.end());
    for(int l = 1, r = n; l < r; ++l, --r){
        while(r > l && v[r] + v[l] < k) ++l;
        if(r > l) ans++;
    }
    cout << ans << '\n';
}
//=====================================
signed main() {
#ifdef LOCAL
    freopen("in.in", "r", stdin);
    freopen("out.out", "w", stdout);
    clock_t c1 = clock();
#endif
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    //=====================================
    int t;
    cin >> t;
    while(t--) solve();
    //=====================================
#ifdef LOCAL
    cerr << "Time used: " << clock() - c1 << " ms" << '\n';
#endif
    return 0;
}

F - Shifting String

题意

给定一个长度为 nnn 的字符串 sss 和数组 ppp，字符串中的每一个字符 sis_isi​ 经过一次变换移动到 pip_ipi​ 位置，问在经过最少几次变换后，该字符串能够与初始字符串相同。

（于是开始手动模拟样例…）

模拟完样例3之后可知每一位上的字符变换都遵循一个循环，也就是说，这个字符串的每一位字符都是以周期若干的环进行滚动的，于是不难发现要做的最少操作数至多就是每个环长度的最小公倍数。（考虑到字符可能相等，所以要判断是否有必要经过一整个周期才可以与初始字符相同）

Code 
#include <iostream>
#include <ctime>
#include <algorithm>
#include <vector>
#include <map>
#include <set>
#define int long long
using namespace std;
void solve() {
    int n, ans = 1;
    string s;
    cin >> n >> s;
    map<int, int> mp;
    for(int i = 0; i < n; ++i){
    int x;
    cin >> x;
    mp[i] = --x;
    }
    vector<bool> vis(n);
    for(int i = 0; i < n; ++i){
        if(vis[i]) continue;
        vis[i] = true;
        int nex = mp[i];
        string tmp = string(1, s[i]);
        while(nex != i){
            vis[nex] = true;
            tmp += s[nex];
            nex = mp[nex];
        }
        int deg = (tmp + tmp).find(tmp, 1);
        ans = ans * deg / __gcd(ans, deg);
    }
    cout << ans << '\n';
}
//=====================================
signed main() {
#ifdef LOCAL
    freopen("in.in", "r", stdin);
    freopen("out.out", "w", stdout);
    clock_t c1 = clock();
#endif
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    //=====================================
    int t;
    cin >> t;
    while(t--) solve();
    //=====================================
#ifdef LOCAL
    cerr << "Time used: " << clock() - c1 << " ms" << '\n';
#endif
    return 0;
}
